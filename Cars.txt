cars
____________________

id - INT - UNSIGNED - PRIMARY_KEY - AUTO_INCREMENT
marca - VARCHAR(50) - NOT_NULL
modello - VARCHAR(50) - NOT_NULL
anno - YEAR - NOT_NULL
prezzo - DECIMAL(11,2) 999.999.999,99 - NOT_NULL
carburante - VARCHAR(30)
cilindrata - FLOAT(7,2) 99.999,99
lunghezza - FLOAT(3,1) 99,9
largezza - FLOAT(2,1) 9,9
altezza - FLOAT(2,1) 9,9
peso - SMALLINT - UNSIGNED
km_percorsi - MEDIUMINT - UNSIGNED - NOT_NULL
condizioni - FLOAT(3,1) 99,9
colori_esterni - VARCHAR(30) - NOT_NULL
colori_interni - VARCHAR(30)
materiale_interno  - VARCHAR(30)
comfort - VARCHAR(250)
targa - VARCHAR(15) - PRIMARY_KEY
concessionario - VARCHAR(50) - NOT_NULL
disponibile - TINYINT -DEFAULT(1)


//AUTO_INCREMENT= AUMENTA IN AUTOMATICO L'ID AUMENTANDO SEMPRE DI 1 PARTENDO DALL'ULTIMO CHE GIA ABBIAMO.
//UNSIGNED= NON HA VALORE NEGATIVO.
//UNIQUE= NON HA RIPETIZIONI DI VALORE NEL DATABASE.
//NOT_NULL= OBBLIGA IL RIEMPIMENTO DEL CAMPO (ALTRIMENTI SE IL CAMPO RIMANE VUOTO METTE NULL DI DEFAULT).
//PRIMARY_KEY= CONTIENE IN AUTOMATICO UNIQUE E NOT NULL.
//DEFAULT= SE L'UTENTE NON INSERISCI NIENTE IMPOSTA UN VALORE/TESTO/IMMAGINE DI PLACEHOLDER.
